## 1. Структура решения

В Visual Studio создайте **Solution** с названием `ReadCity` (Читай город).

```
ReadCity.sln
├── ReadCity.Data/          — Задание 2: библиотека DAL (Class Library)
│   ├── Models/             — классы-модели (Product, Order, User, ...)
│   ├── Context/            — DbContext (EF Core)
│   └── Services/           — сервисы для работы с БД
├── ReadCity.WPF/           — Задание 4–13: оконное приложение (WPF App)
│   ├── Views/              — окна (.xaml)
│   ├── Resources/          — логотип, иконка, изображения
│   └── Helpers/            — вспомогательные классы
└── ReadCity.API/           — Задание 3: Web API (опционально)
```

**Создание проектов:**
1. `Файл → Создать → Проект` → выберите **WPF App (.NET)** → назовите `ReadCity.WPF`
2. ПКМ на Solution → `Добавить → Новый проект` → **Class Library (.NET)** → `ReadCity.Data`
3. В проекте `ReadCity.WPF` добавьте ссылку на `ReadCity.Data`: ПКМ на `Зависимости → Добавить ссылку на проект`

**NuGet пакеты для `ReadCity.Data`:**
```
PM> Install-Package Microsoft.EntityFrameworkCore.SqlServer
PM> Install-Package Microsoft.EntityFrameworkCore.Tools
```

```sql
    StatusId   INT IDENTITY(1,1) PRIMARY KEY,

);
```

```csharp
namespace ReadCity.Data.Models
{
    /// <summary>Товар в каталоге книжного магазина.</summary>
    public class Product
    {
        public int ProductId { get; set; }
        public string ArticleNumber { get; set; } = string.Empty;
        public string ProductName { get; set; } = string.Empty;
        public string Unit { get; set; } = "шт.";

        public string? Author { get; set; }

        public int Discount { get; set; }

        public string? PhotoFileName { get; set; }

        // Навигационные свойства
        public Manufacturer? Manufacturer { get; set; }
        public Category? Category { get; set; }

        /// <summary>Цена с учётом скидки.</summary>
        public decimal DiscountedPrice =>
            Discount > 0 ? Math.Round(Price * (1 - Discount / 100m), 2) : Price;
    }
}
```

```csharp
using Microsoft.EntityFrameworkCore;
using ReadCity.Data.Context;
using ReadCity.Data.Models;

namespace ReadCity.Data.Services
{
    /// <summary>Сервис для работы с товарами.</summary>
    public class ProductService
    {
        /// <summary>Возвращает все товары с загруженными производителями и категориями.</summary>
        public List<Product> GetAllProducts()
        {
            try
            {
                using var context = new ReadCityDbContext();
                return context.Products
                    .Include(p => p.Manufacturer)
                    .Include(p => p.Category)
                    .ToList();
            }
            catch (Exception ex)
            {
                // Логируем ошибку, возвращаем пустой список
                Console.WriteLine($"Ошибка загрузки товаров: {ex.Message}");
                return new List<Product>();
            }
        }
```

```csharp
using Microsoft.EntityFrameworkCore;
using ReadCity.Data.Context;
using ReadCity.Data.Models;

namespace ReadCity.Data.Services
{
    /// <summary>Сервис для работы с пользователями и авторизацией.</summary>
    public class UserService
    {
        /// <summary>
        /// Авторизует пользователя по логину и паролю.
        /// </summary>
        /// <param name="login">Логин пользователя.</param>
        /// <param name="password">Пароль пользователя.</param>
        /// <returns>Объект User или null, если авторизация не удалась.</returns>
        public User? Authenticate(string login, string password)
        {
            try
            {
                using var context = new ReadCityDbContext();
                return context.Users
                    .Include(u => u.Role)
                    .FirstOrDefault(u => u.Login == login && u.Password == password);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Ошибка авторизации: {ex.Message}");
                return null;
            }
        }
    }
}
```

**OrderService.cs**
```csharp
using Microsoft.EntityFrameworkCore;
using ReadCity.Data.Context;
using ReadCity.Data.Models;

namespace ReadCity.Data.Services
{
    /// <summary>Сервис для работы с заказами.</summary>
    public class OrderService
    {
        /// <summary>Сохраняет новый заказ в базу данных.</summary>
        /// <param name="order">Заказ для сохранения.</param>
        /// <returns>True при успехе, false при ошибке.</returns>
        public bool SaveOrder(Order order)
        {
            try
            {
                using var context = new ReadCityDbContext();
                // Следующий номер заказа = максимальный + 1
                order.OrderNumber = (context.Orders.Max(o => (int?)o.OrderNumber) ?? 0) + 1;
                context.Orders.Add(order);
                context.SaveChanges();
                return true;
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Ошибка сохранения заказа: {ex.Message}");
                return false;
            }
        }

        /// <summary>Возвращает все заказы с деталями.</summary>
        public List<Order> GetAllOrders()
        {
            try
            {
                using var context = new ReadCityDbContext();
                return context.Orders
                    .Include(o => o.User)
                    .Include(o => o.Status)
                    .Include(o => o.OrderItems)
                        .ThenInclude(oi => oi.Product)
                    .OrderByDescending(o => o.OrderDate)
                    .ToList();
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Ошибка загрузки заказов: {ex.Message}");
                return new List<Order>();
            }
        }

        /// <summary>Обновляет дату доставки и статус заказа.</summary>
        public bool UpdateOrder(int orderId, DateTime? deliveryDate, int statusId)
        {
            try
            {
                using var context = new ReadCityDbContext();
                var order = context.Orders.Find(orderId);
                if (order == null) return false;

                order.DeliveryDate = deliveryDate;
                order.StatusId = statusId;
                context.SaveChanges();
                return true;
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Ошибка обновления заказа: {ex.Message}");
                return false;
            }
        }
```

```csharp
using ReadCity.Data.Models;
using ReadCity.Data.Services;
using ReadCity.WPF.Views;

namespace ReadCity.WPF.Views
{
    /// <summary>Главное окно просмотра каталога товаров.</summary>
    public partial class ProductListWindow : Window
    {
        private readonly ProductService _productService = new ProductService();
        private List<Product> _allProducts = new List<Product>();

        public ProductListWindow()
        {
            InitializeComponent();
            LoadData();
        }

        /// <summary>Загружает товары и производителей из БД.</summary>
        private void LoadData()
        {
            try
            {
                _allProducts = _productService.GetAllProducts();

                // Заполняем ComboBox производителей
                var manufacturers = _productService.GetManufacturers();
                ManufacturerComboBox.Items.Clear();
                ManufacturerComboBox.Items.Add(new ComboBoxItem
                {
                    Content = "Все производители",
                    Tag = 0
                });
                foreach (var m in manufacturers)
                    ManufacturerComboBox.Items.Add(new ComboBoxItem
                    {
                        Content = m.ManufacturerName,
                        Tag = m.ManufacturerId
                    });
                ManufacturerComboBox.SelectedIndex = 0;

                ApplyFilters();
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Ошибка загрузки данных:\n{ex.Message}",
                    "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }

        /// <summary>Применяет фильтры, поиск и сортировку к списку товаров.</summary>
        private void ApplyFilters()
        {
            try
            {
                var filtered = _allProducts.AsEnumerable();

                // Поиск по названию
                var searchText = SearchTextBox.Text?.Trim().ToLower();
                if (!string.IsNullOrEmpty(searchText))
                    filtered = filtered.Where(p =>
                        p.ProductName.ToLower().Contains(searchText));

                // Фильтр по производителю
                if (ManufacturerComboBox.SelectedItem is ComboBoxItem { Tag: int mId } && mId > 0)
                    filtered = filtered.Where(p => p.ManufacturerId == mId);

                // Фильтр по цене
                if (decimal.TryParse(MinPriceTextBox.Text, out decimal minPrice))
                    filtered = filtered.Where(p => p.DiscountedPrice >= minPrice);
                if (decimal.TryParse(MaxPriceTextBox.Text, out decimal maxPrice))
                    filtered = filtered.Where(p => p.DiscountedPrice <= maxPrice);

                // Сортировка
                var sortTag = (SortComboBox.SelectedItem as ComboBoxItem)?.Tag?.ToString();
                filtered = sortTag switch
                {
                    "NameAsc"   => filtered.OrderBy(p => p.ProductName),
                    "NameDesc"  => filtered.OrderByDescending(p => p.ProductName),
                    "PriceAsc"  => filtered.OrderBy(p => p.DiscountedPrice),
                    "PriceDesc" => filtered.OrderByDescending(p => p.DiscountedPrice),
                    _           => filtered
                };

                var result = filtered.ToList();
                ProductsItemsControl.ItemsSource = result;
                CounterTextBlock.Text = $"Показано: {result.Count} из {_allProducts.Count}";
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Ошибка фильтрации:\n{ex.Message}",
                    "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }

        private void FilterChanged(object sender, EventArgs e) => ApplyFilters();

        /// <summary>Обрабатывает нажатие кнопки «Заказать» для товара.</summary>
        private void OrderButton_Click(object sender, RoutedEventArgs e)
        {
            if (sender is Button { Tag: Product product })
            {
                var existing = App.CurrentOrderItems
                    .FirstOrDefault(i => i.ProductId == product.ProductId);
                if (existing != null)
                    existing.Quantity++;
                else
                    App.CurrentOrderItems.Add(new OrderItem
                    {
                        ProductId = product.ProductId,
                        Product = product,
                        Quantity = 1
                    });

                ViewOrderButton.Visibility = Visibility.Visible;
                MessageBox.Show($"«{product.ProductName}» добавлен в заказ.",
                    "Добавлено", MessageBoxButton.OK, MessageBoxImage.Information);
            }
        }

        private void LoginButton_Click(object sender, RoutedEventArgs e)
        {
            if (App.CurrentUser == null)
            {
                var loginWindow = new LoginWindow();
                loginWindow.ShowDialog();
                UpdateUserPanel();
            }
            else
            {
                App.CurrentUser = null;
                UpdateUserPanel();
            }
        }

        /// <summary>Обновляет панель пользователя после входа/выхода.</summary>
        private void UpdateUserPanel()
        {
            if (App.CurrentUser != null)
            {
                UserNameTextBlock.Text = App.CurrentUser.FullName;
                LoginButton.Content = "Выйти";
                bool isStaff = App.CurrentUser.Role?.RoleName is "Администратор" or "Менеджер";
                OrdersManagementButton.Visibility =
                    isStaff ? Visibility.Visible : Visibility.Collapsed;
            }
            else
            {
                UserNameTextBlock.Text = string.Empty;
                LoginButton.Content = "Войти";
                OrdersManagementButton.Visibility = Visibility.Collapsed;
            }
        }

        private void ViewOrderButton_Click(object sender, RoutedEventArgs e)
        {
            var orderWindow = new OrderWindow();

        }

        private void OrdersManagementButton_Click(object sender, RoutedEventArgs e)

            new OrderManagementWindow().ShowDialog();
        }
    }
```

```csharp
using ReadCity.Data.Models;

namespace ReadCity.WPF.Views

        {

                    OrderDataGrid.ItemsSource = null;
                    OrderDataGrid.ItemsSource = App.CurrentOrderItems;
                    UpdateTotal();
```

## 10. Задание 10: Сохранение талона в TXT

**ReceiptWindow.xaml.cs** (создайте простое окно с TextBlock и кнопкой «Сохранить»):
```csharp
/// <summary>Окно отображения и сохранения талона заказа.</summary>
public partial class ReceiptWindow : Window
{
    private readonly Order _order;
    private readonly List<OrderItem> _items;
    private string _receiptText = string.Empty;

    public ReceiptWindow(Order order, List<OrderItem> items)
    {
        InitializeComponent();
        _order = order;
        _items = items;
        BuildReceipt();
    }

    /// <summary>Формирует текст талона.</summary>
    private void BuildReceipt()
    {
        var sb = new System.Text.StringBuilder();
        sb.AppendLine("=== ТАЛОН ЗАКАЗА ===");
        sb.AppendLine($"Дата заказа:   {_order.OrderDate:dd.MM.yyyy HH:mm}");
        sb.AppendLine($"Номер заказа:  {_order.OrderNumber}");
        sb.AppendLine($"Код получения: {_order.ReceiptCode}");
        sb.AppendLine("-------------------");
        sb.AppendLine("Состав заказа:");
        decimal total = 0;
        foreach (var item in _items)
        {
            var price = item.Product?.DiscountedPrice ?? 0;
            var sum   = price * item.Quantity;
            total += sum;
            sb.AppendLine($"  {item.Product?.ProductName} x{item.Quantity} = {sum:N2} руб.");
        }
        sb.AppendLine("-------------------");
        sb.AppendLine($"ИТОГО: {total:N2} руб.");
        _receiptText = sb.ToString();
        ReceiptTextBlock.Text = _receiptText; // привяжите к TextBlock в XAML
    }

    private void SaveButton_Click(object sender, RoutedEventArgs e)
    {
        var dialog = new Microsoft.Win32.SaveFileDialog
        {
            Filter     = "Текстовые файлы (*.txt)|*.txt",
            FileName   = $"Заказ_{_order.OrderNumber}.txt",
            DefaultExt = "txt"
        };

        if (dialog.ShowDialog() == true)
        {
            try
            {
                File.WriteAllText(dialog.FileName, _receiptText, System.Text.Encoding.UTF8);
                MessageBox.Show("Талон успешно сохранён.",
                    "Готово", MessageBoxButton.OK, MessageBoxImage.Information);
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Не удалось сохранить файл:\n{ex.Message}",
                    "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }
    }
}
```

```csharp
/// <summary>Окно управления заказами (доступно менеджеру и администратору).</summary>

    private readonly OrderService _orderService = new OrderService();

    public OrderManagementWindow()
    {
        InitializeComponent();
        // Доступно только менеджеру и администратору
        if (App.CurrentUser?.Role?.RoleName is not ("Администратор" or "Менеджер"))
        {
            MessageBox.Show("Доступ запрещён.", "Ошибка",
                MessageBoxButton.OK, MessageBoxImage.Error);
            Close();
            return;
        }
        LoadOrders();
    }

    private void LoadOrders()
    {
        try
        {
            OrdersDataGrid.ItemsSource = _orderService.GetAllOrders();
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Ошибка загрузки заказов:\n{ex.Message}",
                "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
        }
    }

    private void SaveOrderButton_Click(object sender, RoutedEventArgs e)
    {
        if (OrdersDataGrid.SelectedItem is not Order selectedOrder)
        {
            MessageBox.Show("Выберите заказ.", "Предупреждение",
                MessageBoxButton.OK, MessageBoxImage.Warning);
            return;
        }

        try
        {
            var statusId = (StatusComboBox.SelectedItem as OrderStatus)?.StatusId
                           ?? selectedOrder.StatusId;
            var deliveryDate = DeliveryDatePicker.SelectedDate;

            var success = _orderService.UpdateOrder(selectedOrder.OrderId, deliveryDate, statusId);
            if (success)
                MessageBox.Show("Заказ обновлён.", "Готово",
                    MessageBoxButton.OK, MessageBoxImage.Information);
            else
                MessageBox.Show("Не удалось обновить заказ.",
                    "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);

            LoadOrders();
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Ошибка при обновлении:\n{ex.Message}",
                "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
        }
    }
}
```
